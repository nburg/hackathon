🧪 Example Commands (After Loading)

// Get all nouns const nouns = P3.filterCandidates( P3.extractCandidates(document), c => c.pos === 'Noun' ); console.log('Nouns:', nouns.map(n => n.word)); // Exclude proper nouns (names/places) const common = P3.filterCandidates( P3.extractCandidates(document), c => !c.isProperNoun ); console.log('Common words:', common.slice(0, 10)); // Get long words (7+ chars) const long = P3.filterCandidates( P3.extractCandidates(document), c => c.word.length >= 7 ); console.log('Long words:', long.map(w => w.word)); // Group by parent element const groups = P3.groupCandidatesByElement( P3.extractCandidates(document) ); console.log(`Words across ${groups.size} elements`);
